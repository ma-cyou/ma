import { error } from '@sveltejs/kit';

export function load({ params }) {
	const usernames = ['john', 'jane', 'alice'];
	const username = params.username;

	if (!usernames.includes(username)) {
		throw error(404, `User ${username} not found`);
	}

	return {
		username
	};
}
